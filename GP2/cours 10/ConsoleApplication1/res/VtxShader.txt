uniform vec4 matrix;
uniform vec4 uvCoord;
uniform vec4 uvPos;


void main()
{
	vec4 worldPos = gl_Vertex;
	worldPos += matrix;
    // transforme la position du vertex
    gl_Position = gl_ModelViewProjectionMatrix * worldPos;

    // transforme les coordonn√©es de texture
    gl_TexCoord[0] = gl_TextureMatrix[0] * (gl_MultiTexCoord0 + uvCoord) * uvPos;

    // transmet la couleur
    gl_FrontColor = gl_Color;
}

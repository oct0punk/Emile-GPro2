{
	vec4 worldPos = gl_Vertex;
	worldPos += matrix;

    gl_Position = gl_ModelViewProjectionMatrix * worldPos;

    // transforme les coordonn√©es de texture
    gl_TexCoord[0] = gl_TextureMatrix[0] * gl_MultiTexCoord0;

}
